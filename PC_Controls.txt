ii's Stupid Menu PC Controls
Q :: Open Menu
\ :: Toggle GUI
E :: A (VR controller)
R :: B (VR controller)
F :: X (VR controller)
G :: Y (VR controller)
[ :: Left Grip
] :: Right Grip
- :: Left Trigger
+ :: Right Trigger
Arrow Keys :: Left Joystick
Left Alt + Arrow Keys :: Right Joystick
End :: Left Joystick Click
Left Alt + End :: Right Joystick Click
